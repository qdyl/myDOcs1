                      /* 样式重置 */
*{
  margin: 0;
  padding: 0;
}
a{
  text-decoration: none;
}
ul,ol,li{
  list-style-type: none;
}

/* 工具类 */

.fleft{
  float:left
}
.fright{
  float: right;
}
.clearfix:before,.clearfix:after{
  content: '';
  display: block;
  height: 0;
  clear: both;
  visibility: hidden;
}
// 单行省略号
.ellipsis{
  overflow: hidden;
  white-space: nowrap;
  text-overflow: ellipsis;
}
.ellipsis2{
  display: -webkit-box;
  -webkit-box-orient: vertical;
  -webkit-line-clamp: 2;
  overflow: hidden;
}
.ellipsis3{
  display: -webkit-box;
  -webkit-box-orient: vertical;
  -webkit-line-clamp: 3;
  overflow: hidden;
}
